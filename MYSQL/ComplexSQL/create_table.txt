CREATE TABLE PROJECT(
	Project_ID INT PRIMARY KEY,
	Title VARCHAR(50),
	Archived BOOL NOT NULL,
	Project_Status VARCHAR(10)
	);
	
CREATE TABLE LANGUAGE_USED(
	Project_ID INT,
	Language VARCHAR(10),
	FOREIGN KEY(Project_ID) REFERENCES PROJECT(Project_ID),
	PRIMARY KEY(Project_ID, Language)
	);
	
CREATE TABLE CONTACT_HISTORY(
	Contact_ID INT,
	Project_ID INT,
	CP_Name VARCHAR(10),
	Contact_Type VARCHAR(10),
	Contact_Details VARCHAR(100),
	FOREIGN KEY(Project_ID) REFERENCES PROJECT(Project_ID),
	PRIMARY KEY(Project_ID, Contact_ID)
	);
	
CREATE TABLE REVIEWER(
	SSN CHAR(9) PRIMARY KEY,
	First_Name VARCHAR(10),
	Last_Name VARCHAR(10)
	);
	
CREATE TABLE REVIEWS(
	Project_ID INT,
	SSN CHAR(9),
	Reviewer_Role VARCHAR(10),
	FOREIGN KEY(Project_ID) REFERENCES PROJECT(Project_ID),
	FOREIGN KEY(SSN) REFERENCES REVIEWER(SSN),
	PRIMARY KEY(Project_ID, SSN)
	);
	
CREATE TABLE INVESTIGATOR(
	SSN CHAR(9) PRIMARY KEY,
	First_Name VARCHAR(10),
	Last_Name VARCHAR(10),
	Organization VARCHAR(10),
	Address VARCHAR(10),
	Phone CHAR(10)
	);
	
CREATE TABLE WORKS_ON(
	Project_ID INT,
	SSN CHAR(9),
	Inv_Role VARCHAR(10),
	FOREIGN KEY(Project_ID) REFERENCES PROJECT(Project_ID),
	FOREIGN KEY(SSN) REFERENCES INVESTIGATOR(SSN),
	PRIMARY KEY(Project_ID, SSN)
	);
